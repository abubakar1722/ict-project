<script>
let budget = 0;
let duration = "";
let expenses = [];
// Navigation to Page 2
function goToPage2() {
budget = document.getElementById("budgetInput").value;
duration = document.getElementById("durationInput").value;
if (budget > 0) {
document.getElementById("page1").classList.remove("active");
document.getElementById("page2").classList.add("active");
} else {
alert("Please enter your budget first!");
}
}
// Add item and calculate total based on frequency
function addItem() {
let name = document.getElementById("nameInput").value;
let cat = document.getElementById("catInput").value;
let cost = parseFloat(document.getElementById("costInput").value);
let freq = document.getElementById("freqInput").value;
if (!name || !cost) { alert("Missing name or cost!"); return; }
let multi = 1;
if (duration === "month") {
let map = {once:1, daily:30, weekdays:22, weekends:8, weekly:4, 'twice-week':8};
multi = map[freq];
} else {
let map = {once:1, daily:7, weekdays:5, weekends:2, weekly:1, 'twice-week':2};
multi = map[freq];
}
let totalCost = cost * multi;
expenses.push({ name, cat, totalCost });
document.getElementById("currentList").innerHTML += "<p>Added: " + name + " (Rs. " + totalCost + ")</p>";
document.getElementById("nameInput").value = "";
document.getElementById("costInput").value = "";
}
// Final Report Logic
function goToPage3() {
document.getElementById("page2").classList.remove("active");
document.getElementById("page3").classList.add("active");
let totalSpent = 0;
expenses.forEach(x => totalSpent += x.totalCost);
let savings = budget - totalSpent;
let exceeded = 0;
// If negative, set savings to 0 and show exceeded row
if (savings < 0) {
exceeded = Math.abs(savings);
savings = 0;
document.getElementById("exceededRow").style.display = "block";
document.getElementById("resExceeded").innerText = "Rs. " + exceeded;
}
document.getElementById("resBudget").innerText = "Rs. " + budget;
document.getElementById("resSavings").innerText = "Rs. " + savings;
let tableBody = document.getElementById("resultTableBody");
expenses.forEach(x => {
let verdict = "";
let vClass = "";
let reason = "";
if (x.cat === "essential") {
verdict = "Very Wise";
vClass = "wise-high";
reason = "Necessity";
} else {
// 5,000 PKR Rule for Non-Essentials
if (duration === "month" && x.totalCost > 5000) {
verdict = "Not Wise";
vClass = "unwise-red";
reason = "Luxury > 5000 Rs.";
} else if (x.totalCost > (budget * 0.15)) {
verdict = "Caution";
vClass = "caution";
reason = "High Budget %";
} else {
verdict = "Fair Choice";
vClass = "wise-med";
reason = "Manageable";
}
}
tableBody.innerHTML += <tr><td>${x.name}</td><td class="${vClass}">${verdict}</td><td>${reason}</td></tr>;
});
// Weekly Advice
if (duration === "month" && savings > 0) {
document.getElementById("adviceArea").innerHTML = "<b>üí° Advice:</b> You have extra money! Your weekly safe limit is <b>Rs. " + (savings/4).toFixed(0) + "</b>.";
} else if (exceeded > 0) {
document.getElementById("adviceArea").innerHTML = "<b>üí° Advice:</b> You are spending more than you have. Focus on 'Essential' items only.";
}
}
